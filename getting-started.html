Getting started

Installing and using Formatic.

Setup

You can import Formatic as a CommonJS module or as a global.

CommonJS

$ npm install formatic

// Get the Formatic class.
var Formatic = require('formatic');

// Create some fields.
var fields = [
  {
    type: 'string',
    isSingleLine: true,
    key: 'firstName',
    label: 'First Name'
  },
  {
    type: 'str',
    isSingleLine: true,
    key: 'lastName',
    label: 'Last Name'
  }
];

// Render the form.
React.render(React.createElement(Formatic, {
  fields: fields,
  onChange: function (newValue) {
    console.log(newValue);
  }
}), document.body);

Global

$ npm install formatic
