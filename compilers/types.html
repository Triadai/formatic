compilers.types

Convert some high-level types to low-level types and properties.

'use strict'; var _ = require('underscore'); module.exports = function (plugin) {

Map high-level type to low-level type. If a function is supplied, can modify the field definition.

var typeCoerce = { unicode: function (def) { def.type = 'string'; def.maxRows = 1; }, text: 'string', select: function (def) { def.choices = def.choices || []; }, bool: 'boolean' }; typeCoerce.str = typeCoerce.unicode; plugin.exports.compile = function (def) { var coerceType = typeCoerce[def.type]; if (coerceType) { if (_.isString(coerceType)) { def.type = coerceType; } else if (_.isFunction(coerceType)) { def = coerceType(def); } } }; };
