core.form-init

This plugin makes it easy to hook into form initialization, without having to configure all the other core plugins.

'use strict'; module.exports = function (plugin) { var initPlugins = plugin.requireAll(plugin.config.init); var proto = plugin.exports; proto.init = function () { var form = this; initPlugins.forEach(function (plugin) { plugin.apply(form, arguments); }); }; };
