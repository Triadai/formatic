object-item-control component

Render the remove buttons for an object item.

'use strict'; var React = require('react'); var R = React.DOM; var cx = React.addons.classSet; module.exports = React.createClass({ displayName: 'ObjectItemControl', mixins: [require('../../mixins/helper')], onRemove: function () { this.props.onRemove(this.props.itemKey); }, render: function () { return this.renderWithConfig(); }, renderDefault: function () { var config = this.props.config; var field = this.props.field; return R.div({className: cx(this.props.classes)}, config.createElement('remove-item', {field: field, onClick: this.onRemove}) ); } });
