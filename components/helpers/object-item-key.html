component.object-item-key

Render an object item key editor.

'use strict'; var React = require('react'); var R = React.DOM; var cx = React.addons.classSet; module.exports = React.createClass({ displayName: 'ObjectItemKey', mixins: [require('../../mixins/helper')], onChange: function (event) { this.props.onChange(event.target.value); }, render: function () { return this.renderWithConfig(); }, renderDefault: function () { return R.input({className: cx(this.props.className), type: 'text', value: this.props.displayKey, onChange: this.onChange}); } });
