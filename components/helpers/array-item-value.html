array-item-value component

Render the value of an array item.

'use strict'; var React = require('react'); var R = React.DOM; var cx = React.addons.classSet; module.exports = React.createClass({ displayName: 'ArrayItemValue', mixins: [require('../../mixins/helper')], onChangeField: function (newValue, info) { this.props.onChange(this.props.index, newValue, info); }, render: function () { return this.renderWithConfig(); }, renderDefault: function () { var config = this.props.config; var field = this.props.field; return R.div({className: cx(this.props.classes)}, config.createFieldElement({field: field, onChange: this.onChangeField, onAction: this.onBubbleAction}) ); } });
