'use strict'; var React = require('react/addons'); var R = React.DOM; module.exports = function (plugin) { plugin.exports = React.createClass({ mixins: [require('./mixins/field')], getDefaultProps: function () { return { className: plugin.config.className }; }, onChange: function (event) { var newValue = event.target.value; this.props.field.val(newValue); }, render: function () { var field = this.props.field; return plugin.component('field')({ field: field }, R.input({ className: this.props.className, type: 'text', value: field.value, rows: field.def.rows, onChange: this.onChange })); } }); };
