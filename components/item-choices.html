component.item-choices

Give a list of choices of item types to create as children of an field.

'use strict'; var React = require('react/addons'); var R = React.DOM; module.exports = function (plugin) { plugin.exports = React.createClass({ getDefaultProps: function () { return { className: plugin.config.className }; }, onChange: function (event) { this.props.onSelect(parseInt(event.target.value)); }, render: function () { var field = this.props.field; var typeChoices = null; if (field.items().length > 0) { typeChoices = R.select({className: this.props.className, value: this.value, onChange: this.onChange}, field.items().map(function (item, i) { return R.option({key: i, value: i}, item.label || i); }) ); } return typeChoices ? typeChoices : R.span(null); } }); };
