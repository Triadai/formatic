'use strict'; var React = require('react/addons'); var R = React.DOM; module.exports = function (plugin) { plugin.exports = React.createClass({ getInitialState: function () { return { field: this.props.form.field() }; }, componentDidMount: function() { var form = this.props.form; form.on('change', this.onFormChanged); }, componentWillUnmount: function () { var form = this.props.form; form.off('change', this.onFormChanged); }, onFormChanged: function () { if (this.props.onChange) { this.props.onChange(this.props.form.val()); } this.setState({ field: this.props.form.field() }); }, render: function () { return R.div({className: 'formatic'}, this.state.field.component() ); } }); };
